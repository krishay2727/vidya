#### 5C — JS function to load component data
```javascript
async function loadComponentCard(id) {
  try {
    const res = await fetch(`data/components/${id}.json`);
    if (!res.ok) return null;
    return await res.json();
  } catch(e) {
    return null;
  }
}

async function renderComponentsTab(project) {
  const container = document.getElementById('ptab-components');
  if (!container) return;
  const refs = project.componentRefs || [];
  if (!refs.length) {
    container.innerHTML = renderProjectEmptyState('🧩', 'No Component Details', 'Component data files not added yet.');
    return;
  }
  const cards = await Promise.all(refs.map(id => loadComponentCard(id)));
  container.innerHTML = cards
    .filter(Boolean)
    .map(c => buildComponentCardHTML(c))
    .join('');
}
```
